Basics

Writing out a file to disk

package main

import (
        "io/ioutil"
)

func main() {
        fileName := "sample.txt"
        fileContents := []byte("thisIsASimpleFile")
        ioutil.WriteFile(fileName, fileContents, 0600)
}
