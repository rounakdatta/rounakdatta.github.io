Apache Livy

Hadoop

Apache Livy is a project which provides a REST interface to submit a Spark job as well as query the status of it. The spark-submit configurations can be put together in the format of a JSON and submitted through a cURL request.

How to submit using Livy

  1. Using cURL
    curl -H 'X-Requested-By: admin' -H 'Content-Type: application/json' -X POST -d @deployment-config.json 10.6.34.194:8999/batches
  2. Using a Python wrapper
    livy_submitter.py
    import requests
import sys
import json
import logging 
import time

LIVY_URL="core.data.hotstar-labs.com"

def setup_logger():
    """
    A utility function to setup the logging messages to stdout.
    :return:
    """
    root = logging.getLogger()
    root.setLevel(logging.INFO)
    ch = logging.StreamHandler(sys.stderr)
    ch.setLevel(logging.INFO)
    formatter = logging.Formatter(
        '%(asctime)s - %(module)s - %(levelname)s - %(message)s')
    ch.setFormatter(formatter)
    root.addHandler(ch)

def poll_till_completion(batch_id):
    """
    Poll a livy job till it completes
    """
    LIVY_STATUS_ENDPOINT = "http://{0}:8999/batches/{1}/state".format(LIVY_URL, batch_id)
    response = requests.get(LIVY_STATUS_ENDPOINT)
    status = response.json()["state"]
    while (status in ['starting', 'running']):
        logging.info("The current status for batch {0} is {1}".format(batch_id, status))
        time.sleep(30)
        response = requests.get(LIVY_STATUS_ENDPOINT)
        status = response.json()["state"]
    if status != "success":
        sys.exit(-1)  

def launch_spark_job(payload):
    """
    Launch a spark job and poll till completion.
    """
    headers = {
        'content-type': 'application/json',
        'X-Requested-By': 'admin'
    }
    LIVY_JOB_ENDPOINT = "http://{0}:8999/batches".format(LIVY_URL)
    response = requests.post(LIVY_JOB_ENDPOINT, data=json.dumps(payload), headers=headers)
    response.raise_for_status()
    batch_id = response.json()['id']
    poll_till_completion(batch_id)

if __name__ == "__main__":
    setup_logger()
    logging.info("Starting the submission via livy")
    if len(sys.argv) < 2:
        logging.error("The config file argument is not passed.")
    file_name = sys.argv[1]
    with open(file_name, 'r') as handler:
        payload = json.loads(handler.read())
        launch_spark_job(payload)
